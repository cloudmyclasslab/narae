📊 유지보수 신청 → Google Sheets 연동

홈페이지에서 접수된 유지보수 신청이 Google Sheets에 자동으로 기록됩니다.

Apps Script 설정 (5단계)
1
Google Sheets 새 시트 생성 → 시트 이름: 유지보수신청
2
Sheets 메뉴 → 확장 프로그램 → Apps Script
3
아래 코드 붙여넣기 후 저장 
function doPost(e) {
  try {
    const ss = SpreadsheetApp.getActiveSpreadsheet();
    let sh = ss.getSheetByName('유지보수신청');
    if (!sh) {
      sh = ss.insertSheet('유지보수신청');
      sh.appendRow(['접수번호','일자','유지보수 신청 내용','신청인','장소','기기명','접수 시각']);
      sh.setFrozenRows(1);
      sh.getRange(1,1,1,7).setFontWeight('bold').setBackground('#f4b8b5');
      [60,110,380,90,140,120,150].forEach((w,i)=>sh.setColumnWidth(i+1,w));
    }
    // no-cors 방식: postData.contents 로 수신
    const raw = e.postData ? e.postData.contents : e.parameter.data;
    const d = JSON.parse(raw);
    const num = sh.getLastRow(); // 헤더 포함 행 수 = 접수 순번
    sh.appendRow([num, d.date, d.content, d.name, d.location, d.device,
      new Date().toLocaleString('ko-KR')]);
    return ContentService
      .createTextOutput(JSON.stringify({ok:true}))
      .setMimeType(ContentService.MimeType.JSON);
  } catch(err) {
    return ContentService
      .createTextOutput(JSON.stringify({ok:false, error: err.message}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

// ※ 배포 설정: 웹 앱 → 액세스 = '모든 사용자' (익명 포함)

💻 정보화기기 대여 신청 → Google Sheets 연동

정보화기기 대여 신청이 Google Sheets에 자동 기록됩니다. Sheets 열 구성: 일자 | 항목(담당자) | 기기명 | 개수 | 대여자 | 접수 시각

1
Google Sheets에서 시트 이름: 정보화기기대여 생성
2
Apps Script → 아래 코드 사용 (유지보수와 같은 스크립트에 함수 추가 가능) 
function doPost(e) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const d = JSON.parse(e.postData.contents);
  const shName = d.sheet || '유지보수신청';
  let sh = ss.getSheetByName(shName);
  if (!sh) {
    sh = ss.insertSheet(shName);
    if(shName === '정보화기기대여') {
      sh.appendRow(['일자','항목(담당자)','기기명','개수','대여자','접수 시각']);
      sh.getRange(1,1,1,6).setFontWeight('bold').setBackground('#b7d7a8');
    } else {
      sh.appendRow(['접수번호','일자','유지보수 신청 내용','신청인','장소','기기명','접수 시각']);
      sh.getRange(1,1,1,7).setFontWeight('bold').setBackground('#f4b8b5');
    }
    sh.setFrozenRows(1);
  }
  if(shName === '정보화기기대여') {
    sh.appendRow([d.date, d.dept, d.device, d.qty, d.borrower,
      new Date().toLocaleString('ko-KR')]);
  } else {
    const num = sh.getLastRow();
    sh.appendRow([num, d.date, d.content, d.name, d.location, d.device,
      new Date().toLocaleString('ko-KR')]);
  }
  return ContentService.createTextOutput(
    JSON.stringify({ok:true})).setMimeType(ContentService.MimeType.JSON);
}

배포 → 새 배포 → 웹 앱 → 모든 사용자 → URL 복사
